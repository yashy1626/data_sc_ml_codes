style.use('ggplot')

# Scatter plot
# A scatter plot is a diagram where each value in the data set is represented by a dot.
x = [1,2,3,4,5,6,7,8]
y = [5,2,4,2,1,4,5,2]
plt.scatter(x,y,label='scatter',color='g',s=25,marker='v')
plt.title('Scatter Graph')
plt.legend()
plt.ylabel('Y Axis')
plt.xlabel('X Axis')
plt.show()

# Pie Plot
slices = [7,2,2,13]
activities = ['Sleeping','Eating','Playing','Working']
cols = ['c','m','r','y']
plt.pie(slices,
        labels=activities,
        colors=cols,
        startangle=90,
        shadow=True,
        explode=(0,0.1,0,0),
        autopct='%1.1f%%')  # overlay percentage over graph
plt.show()
